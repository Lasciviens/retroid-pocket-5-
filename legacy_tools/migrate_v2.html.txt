// ── API HELPERS ──────────────────────────────────────────────
async function api(endpoint, method = 'GET', body = null) {
  const opts = {
    method,
    headers: {
      'Content-Type': 'application/json',
      'apikey': KEY,
      'Authorization': `Bearer ${KEY}`,
      'Prefer': 'return=representation'
    }
  };
  if (body) opts.body = JSON.stringify(body);
  const res = await fetch(`${URL}/rest/v1/${endpoint}`, opts);
  if (!res.ok) {
    const err = await res.text();
    throw new Error(`${res.status}: ${err}`);
  }
  const text = await res.text();
  return text ? JSON.parse(text) : [];
}

    const sysMap = {};
    sysResult.forEach(s => { sysMap[s.name] = s.id; });
    log(`✓ ${sysResult.length} sistem bulundu`, 'ok');
    setProgress(10);

    // STEP 2: Genres — sadece oku
    log('<span class="step">═══ ADIM 2/5: Türler okunuyor ═══</span>');
    const genResult = await api('genres?select=id,name');
    const genMap = {};
    genResult.forEach(g => { genMap[g.name] = g.id; });
    log(`✓ ${genResult.length} tür bulundu`, 'ok');
    setProgress(20);

    // STEP 3: Series — sadece oku
    log('<span class="step">═══ ADIM 3/5: Seriler okunuyor ═══</span>');
    const serResult = await api('series?select=id,name');
    const serMap = {};
    serResult.forEach(s => { serMap[s.name] = s.id; });
    log(`✓ ${serResult.length} seri bulundu`, 'ok');
    setProgress(30);

    // STEP 4: Emulators — sadece oku
    log('<span class="step">═══ ADIM 4/5: Emülatörler okunuyor ═══</span>');
    const emuResult = await api('emulators?select=id,name');
    const emuMap = {};
    emuResult.forEach(e => { emuMap[e.name] = e.id; });
    log(`✓ ${emuResult.length} emülatör bulundu`, 'ok');
    setProgress(40);

    // STEP 5: Games + platforms + genres
    log('<span class="step">═══ ADIM 5/5: Oyunlar ═══</span>');
    let gameOk = 0, gameFail = 0;
    const total = GAMES.length;

    for (let i = 0; i < GAMES.length; i++) {
      const g = GAMES[i];
      try {
        // Insert game
        const [game] = await api('games', 'POST', [{
          title: g.title,
          series_id: g.series ? serMap[g.series] : null,
          release_year: g.year,
          description: g.desc,
          is_coop: g.coop,
          coop_notes: g.coopNotes || null,
          play_status: 'backlog',
          rating: g.rating,
          is_iconic: g.iconic,
        }]);

        // Insert game_genres
        if (g.genres && g.genres.length > 0) {
          const ggRows = g.genres.filter(gn => genMap[gn]).map(gn => ({ game_id: game.id, genre_id: genMap[gn] }));
          if (ggRows.length > 0) await api('game_genres', 'POST', ggRows);
        }

        // Insert game_platforms
        for (const p of g.platforms) {
          await api('game_platforms', 'POST', [{
            game_id: game.id,
            system_id: sysMap[p.system],
            emulator_id: emuMap[p.emulator] || null,
            performance: p.perf || 'good',
            performance_notes: p.perfNotes || null,
            cover_url: p.cover || null,
            region: 'USA',
          }]);
        }

        gameOk++;
        log(`✓ [${i+1}/${total}] ${g.title}`, 'ok');
      } catch(e) {
        gameFail++;
        log(`✗ [${i+1}/${total}] ${g.title}: ${e.message}`, 'err');
      }

      setProgress(40 + Math.round((i+1)/total * 60));
      await new Promise(r => setTimeout(r, 50));
    }

    log('', '');
    log(`🎉 Migration tamamlandı! ${gameOk} oyun başarılı, ${gameFail} hatalı.`, gameOk > 0 ? 'ok' : 'err');
    setProgress(100);

  } catch(e) {
    log(`💥 Kritik hata: ${e.message}`, 'err');
  }

  document.getElementById('btn').disabled = false;
}
